    <!-- Upload -->

    <div class="mb-6">

      <label class="block mb-2 font-medium">Unggah Foto Produk Dulu</label>

      <input type="file" id="product-image" accept="image/*" class="w-full text-sm text-gray-400 file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:font-semibold file:bg-purple-600 file:text-white hover:file:bg-purple-700">

      <div id="preview" class="mt-4 flex justify-center items-center h-40 bg-gray-700 rounded-lg overflow-hidden">

        <span class="text-gray-500 text-sm">Pratinjau Foto Produk</span>

      </div>

    </div>


    <!-- Generate button -->

    <button id="generate-button" class="w-full bg-purple-600 py-3 rounded-lg font-bold hover:bg-purple-700 transition-colors">

      Generate Konten

    </button>


    <!-- Hasil -->

    <div class="mt-8">

      <h2 class="text-xl font-semibold text-purple-400 mb-4">Hasil Gambar</h2>

      <div id="output" class="grid grid-cols-1 sm:grid-cols-2 gap-4"></div>

    </div>

  </div>


  <script>

    const productInput = document.getElementById('product-image');

    const preview = document.getElementById('preview');

    const generateButton = document.getElementById('generate-button');

    const output = document.getElementById('output');


    productInput.addEventListener('change', (e) => {

      const file = e.target.files[0];

      if (file) {

        const reader = new FileReader();

        reader.onload = (ev) => {

          preview.innerHTML = `<img src="${ev.target.result}" class="h-full object-contain">`;

        };

        reader.readAsDataURL(file);

      }

    });


    generateButton.addEventListener('click', () => {

      output.innerHTML = '';

      for (let i = 0; i < 3; i++) {

        const div = document.createElement('div');

        div.className = "h-40 bg-gray-700 rounded-lg flex items-center justify-center";

        div.innerHTML = `<span class="text-gray-400">[Generated ${i+1}]</span>`;

        output.appendChild(div);

      }

    });

  </script>
